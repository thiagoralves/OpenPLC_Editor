FUNCTION_BLOCK PWM_CONTROLLER
VAR_INPUT
CHANNEL : SINT;
FREQ : REAL;
DUTY : REAL;
END_VAR
VAR_OUTPUT
SUCCESS : BOOL;
END_VAR
VAR
internal_ch : SINT;
internal_freq : REAL;
internal_duty : REAL;
END_VAR

  IF CHANNEL < 1 THEN
    SUCCESS := FALSE;
    RETURN;
  END_IF;


  IF (CHANNEL <> internal_ch) OR (FREQ <> internal_freq) OR (DUTY <> internal_duty) THEN
    SUCCESS := TRUE;
  END_IF;

END_FUNCTION_BLOCK
